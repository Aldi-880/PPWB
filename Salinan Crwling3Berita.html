import requests
from bs4 import BeautifulSoup
import nltk

# Fungsi untuk melakukan crawling berita dari suatu URL
def crawl_berita(url):
    response = requests.get(url)
    html = response.text
    soup = BeautifulSoup(html, 'html.parser')
    judul = soup.find('h1', class_="post-title")
    article = soup.find('div', class_="post-content clearfix")  # Sesuaikan dengan struktur HTML halaman web berita
    if judul and article:
        judul_berita = judul.get_text()
        article_text = article.get_text()
        return judul_berita, article_text
    else:
        return None, None

# Daftar URL dari tiga situs berita yang ingin Anda crawling
urls = [
    "https://www.antaranews.com/berita/3855282/presiden-jokowi-tunjuk-marthinus-hukom-jadi-kepala-bnn",
    "https://www.antaranews.com/berita/3853461/dejan-gloria-paparkan-kunci-juarai-turnamen-super-300-perdana-di-india?utm_source=antaranews&utm_medium=desktop&utm_campaign=menu_news",
    "https://www.antaranews.com/berita/3853809/sasa-ramaikan-event-otomotif-decemblar-kustom-weekender-di-ancol?utm_source=antaranews&utm_medium=desktop&utm_campaign=menu_news"
]

# Loop melalui setiap URL, melakukan crawling, dan tokenisasi kalimat
for url in urls:
    judul_berita, article_text = crawl_berita(url)

    if judul_berita and article_text:
        sentences = nltk.sent_tokenize(article_text)

        # Cetak judul dan kalimat-kalimat dari satu berita
        print(f"Judul Berita dari {url}: {judul_berita}")
        print("Isi Berita:")
        for sentence in sentences:
            print(sentence)
        print("\n")
    else:
        print(f"Gagal mengambil berita dari {url}\n")
